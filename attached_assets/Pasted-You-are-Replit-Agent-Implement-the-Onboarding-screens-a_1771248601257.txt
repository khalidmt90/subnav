```tsx
import React, { useEffect } from 'react';
import { View, StyleSheet } from 'react-native';
import Svg, {
  Defs, LinearGradient, RadialGradient, Stop,
  Filter, FeGaussianBlur, FeMerge, FeMergeNode,
  Ellipse, Path, Line, Circle, G,
} from 'react-native-svg';
import Animated, {
  useSharedValue, useAnimatedStyle, useAnimatedProps,
  withRepeat, withTiming, withSequence,
  Easing, interpolate,
} from 'react-native-reanimated';

const AnimatedG = Animated.createAnimatedComponent(G);
const AnimatedCircle = Animated.createAnimatedComponent(Circle);

interface AppLogoProps {
  size?: number; // logo card size, default 96
  showHalo?: boolean; // show outer glow halo, default true
}

export function AppLogo({ size = 96, showHalo = true }: AppLogoProps) {
  // ── SWEEP ANIMATION ──
  const sweepAngle = useSharedValue(-40);
  useEffect(() => {
    sweepAngle.value = withRepeat(
      withTiming(320, { duration: 3000, easing: Easing.linear }),
      -1,
      false
    );
  }, []);

  // ── HALO PULSE ──
  const haloScale = useSharedValue(1);
  useEffect(() => {
    haloScale.value = withRepeat(
      withSequence(
        withTiming(1.15, { duration: 1500, easing: Easing.inOut(Easing.sine) }),
        withTiming(1,    { duration: 1500, easing: Easing.inOut(Easing.sine) })
      ),
      -1,
      false
    );
  }, []);

  // ── PING DOTS ──
  const ping1 = useSharedValue(0);
  const ping2 = useSharedValue(0);
  const ping3 = useSharedValue(0);

  useEffect(() => {
    const makePing = (sv: Animated.SharedValue<number>, delay: number) => {
      setTimeout(() => {
        sv.value = withRepeat(
          withSequence(
            withTiming(1, { duration: 300 }),
            withTiming(0, { duration: 600 }),
            withTiming(0, { duration: 2100 }) // wait for next cycle
          ),
          -1,
          false
        );
      }, delay);
    };
    makePing(ping1, 0);
    makePing(ping2, 1000);
    makePing(ping3, 2000);
  }, []);

  // Animated styles
  const haloStyle = useAnimatedStyle(() => ({
    transform: [{ scale: haloScale.value }],
    opacity: interpolate(haloScale.value, [1, 1.15], [0.8, 1]),
  }));

  const sweepAnimProps = useAnimatedProps(() => ({
    rotation: sweepAngle.value,
    origin: `29, 42`,
  }));

  const dot1Props = useAnimatedProps(() => ({
    opacity: ping1.value,
    r: interpolate(ping1.value, [0, 1], [1.5, 2.5]),
  }));
  const dot2Props = useAnimatedProps(() => ({
    opacity: ping2.value,
    r: interpolate(ping2.value, [0, 1], [1.5, 2.5]),
  }));
  const dot3Props = useAnimatedProps(() => ({
    opacity: ping3.value,
    r: interpolate(ping3.value, [0, 1], [1.2, 2]),
  }));

  const svgSize = size * 0.604; // 58/96
  const cardRadius = size * 0.27; // 26/96

  return (
    <View style={[styles.container, { width: size + 40, height: size + 40 }]}>
      {/* Outer halo glow */}
      {showHalo && (
        <Animated.View style={[styles.halo, { width: size + 40, height: size + 40, borderRadius: (size + 40) / 2 }, haloStyle]} />
      )}

      {/* Glass card */}
      <View style={[
        styles.card,
        {
          width: size,
          height: size,
          borderRadius: cardRadius,
        }
      ]}>
        {/* Top sheen overlay */}
        <View style={[styles.sheen, { borderRadius: cardRadius }]} />
        {/* Corner reflection */}
        <View style={styles.cornerReflect} />

        {/* SVG Icon */}
        <Svg
          width={svgSize}
          height={svgSize}
          viewBox="0 0 58 58"
          style={{ zIndex: 1 }}
        >
          <Defs>
            <LinearGradient id="arcGrad" x1="0%" y1="0%" x2="100%" y2="100%">
              <Stop offset="0%"   stopColor="#5B8BF8" />
              <Stop offset="50%"  stopColor="#2DD4BF" />
              <Stop offset="100%" stopColor="#38BDF8" />
            </LinearGradient>
            <LinearGradient id="sweepGrad" x1="50%" y1="0%" x2="100%" y2="100%">
              <Stop offset="0%"   stopColor="#5B6CF8" stopOpacity="0.9" />
              <Stop offset="100%" stopColor="#2DD4BF" stopOpacity="0.2" />
            </LinearGradient>
            <RadialGradient id="centerGlow" cx="50%" cy="72%" r="40%">
              <Stop offset="0%"   stopColor="#5B6CF8" stopOpacity="0.6" />
              <Stop offset="100%" stopColor="#5B6CF8" stopOpacity="0" />
            </RadialGradient>
          </Defs>

          {/* Center glow */}
          <Ellipse cx="29" cy="42" rx="18" ry="10" fill="url(#centerGlow)" opacity="0.5" />

          {/* ── 3 CONCENTRIC RADAR ARCS ── */}
          {/* Outer — faintest */}
          <Path
            d="M 14 42 A 18 18 0 0 1 32 24"
            stroke="url(#arcGrad)" strokeWidth="1.2"
            strokeLinecap="round" fill="none" opacity="0.25"
          />
          {/* Mid */}
```

### Full component structure:

```tsx
import React, { useRef, useState, useCallback } from 'react';
import {
  View, Text, FlatList, Pressable, StyleSheet,
  Dimensions, StatusBar, Platform,
} from 'react-native';
import Animated, {
  useSharedValue, useAnimatedStyle, withTiming,
  interpolate, Extrapolation,
} from 'react-native-reanimated';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { LinearGradient } from 'expo-linear-gradient';
import Svg, { Defs, LinearGradient as SvgGrad, Stop, Path, Circle, Line, G } from 'react-native-svg';
import { AppLogo } from '../components/AppLogo';
import { Colors, FontSize, Radii } from '../theme/tokens';
import { useI18n } from '../hooks/useI18n'; // your i18n hook

const { width: W, height: H } = Dimensions.get('window');

export function OnboardingScreen({ navigation }: any) {
  const [currentIndex, setCurrentIndex] = useState(0);
  const flatRef = useRef<FlatList>(null);
  const { lang, t } = useI18n();

  // ── BACKGROUND PARTICLE POSITIONS (static, declared once) ──
  const PARTICLES = [
    { left: '15%', top: '20%', size: 1.5, color: 'rgba(45,212,191,0.5)',  duration: 8000,  delay: 0    },
    { left: '75%', top: '15%', size: 2,   color: 'rgba(91,108,248,0.6)',  duration: 10000, delay: 1500 },
    { left: '35%', top: '35%', size: 1.5, color: 'rgba(91,108,248,0.5)',  duration: 7000,  delay: 3000 },
    { left: '85%', top: '40%', size: 2,   color: 'rgba(45,212,191,0.4)',  duration: 9000,  delay: 500  },
    { left: '55%', top: '18%', size: 1,   color: 'rgba(91,108,248,0.4)',  duration: 11000, delay: 2000 },
    { left: '22%', top: '55%', size: 1.5, color: 'rgba(139,92,246,0.5)', duration: 8000,  delay: 4000 },
    { left: '68%', top: '28%', size: 1.5, color: 'rgba(91,108,248,0.5)', duration: 12000, delay: 1000 },
    { left: '45%', top: '45%', size: 2,   color: 'rgba(45,212,191,0.35)',duration: 9000,  delay: 2500 },
  ];

  const handleNext = useCallback(async () => {
    if (currentIndex < SLIDES.length - 1) {
      const next = currentIndex + 1;
      flatRef.current?.scrollToIndex({ index: next, animated: true });
      setCurrentIndex(next);
    } else {
      // Last slide — mark onboarding done, navigate to Auth
      await AsyncStorage.setItem('onboarding_shown', 'true');
      navigation.replace('Auth');
    }
  }, [currentIndex, navigation]);

  const handleSkip = useCallback(async () => {
    await AsyncStorage.setItem('onboarding_shown', 'true');
    navigation.replace('Auth');
  }, [navigation]);

  const renderSlide = ({ item, index }: { item: typeof SLIDES[0], index: number }) => (
    <View style={{ width: W, height: H }}>
      {/* ── BACKGROUND ── */}

      {/* Layer 1: Deep space base gradient */}
      <LinearGradient
        colors={['#1A1040', '#0D0E1A', '#0B0C14']}
        locations={[0, 0.45, 1]}
        start={{ x: 0.5, y: 0 }}
        end={{ x: 0.5, y: 1 }}
        style={StyleSheet.absoluteFill}
      />

      {/* Layer 2: Aurora glow — two radial glows */}
      {/* Simulate with two positioned Views using shadowColor */}
      <View style={[StyleSheet.absoluteFill, styles.auroraLeft]} />
      <View style={[StyleSheet.absoluteFill, styles.auroraRight]} />

      {/* Layer 3: Dot grid — use a subtle repeating pattern via SVG */}
      <DotGrid />

      {/* Layer 4: Expanding rings at bottom */}
      <ExpandingRings />

      {/* Floating particles */}
      {PARTICLES.map((p, i) => (
        <FloatingParticle key={i} {...p} />
      ))}

      {/* ── SLIDE CONTENT ── */}
      <View style={styles.slideContent}>

        {/* Logo zone */}
        <View style={styles.logoZone}>
          {item.icon === 'radar'  && <AppLogo size={96} showHalo />}
          {item.icon === 'bell'   && <BellIcon size={96} />}
          {item.icon === 'shield' && <ShieldIcon size={96} />}
        </View>

        {/* Text zone */}
        <View style={styles.textZone}>
          <Text style={styles.slideTitle}>
            {typeof item.title === 'string' ? item.title : item.title[lang]}
          </Text>
          <SlideTagline slide={item} lang={lang} />
        </View>

      </View>
    </View>
  );

  return (
    <View style={styles.screen}>
      <StatusBar barStyle="light-content" translucent backgroundColor="transparent" />

      <FlatList
        ref={flatRef}
        data={SLIDES}
        renderItem={renderSlide}
        keyExtractor={s => s.id}
        horizontal
        pagingEnabled
        showsHorizontalScrollIndicator={false}
        scrollEnabled={false} // control via buttons only
        getItemLayout={(_, index) => ({ length: W, offset: W * index, index })}
      />

      {/* ── SKIP BUTTON — top left in RTL context = visually left ── */}
      <Pressable style={styles.skipBtn} onPress={handleSkip}>
        <Text style={styles.skipText}>{lang === 'ar' ? 'تخطي' : 'Skip'}</Text>
      </Pressable>

      {/* ── LANGUAGE TOGGLE — top right ── */}
      <LanguageToggle />

      {/* ── BOTTOM CONTROLS — above content ── */}
      <View style={styles.bottomControls}>

        {/* Progress dots */}
        <View style={styles.dotsRow}>
          {SLIDES.map((_, i) => (
            <View
              key={i}
              style={[
                styles.dot,
                i === currentIndex ? styles.dotActive : styles.dotInactive,
              ]}
            />
          ))}
        </View>

        {/* CTA Button */}
        <Pressable onPress={handleNext}>
          <LinearGradient
            colors={
              currentIndex === 2
                ? ['#059669', '#10B981', '#2DD4BF']
                : ['#5B6CF8', '#7C5CF8', '#6B5CF8']
            }
            start={{ x: 0, y: 0 }}
            end={{ x: 1, y: 1 }}
            style={styles.ctaBtn}
          >
            {/* Arrow box */}
            <View style={styles.ctaArrowBox}>
              <Svg width="14" height="14" viewBox="0 0 14 14" fill="none">
                <Path
                  d="M10 7H4M4 7L7 4M4 7L7 10"
                  stroke="white" strokeWidth="1.8"
                  strokeLinecap="round" strokeLinejoin="round"
                />
              </Svg>
            </View>
            <Text style={styles.ctaText}>
              {lang === 'ar' ? SLIDES[currentIndex].cta.ar : SLIDES[currentIndex].cta.en}
            </Text>
          </LinearGradient>
        </Pressable>

      </View>
    </View>
  );
}
```

### DotGrid — subtle dot pattern over background
```tsx
function DotGrid() {
  // Render a grid of 1px dots using SVG pattern
  // Cover full screen with dots spaced 28px apart
  // Dot color: rgba(255,255,255,0.06)
  // Apply radial mask: dots visible in center, fade to transparent at edges
  // Implement using react-native-svg with a <Defs><Pattern> approach
  // or simply render a View with a repeating pattern via backgroundImage (Expo web)
  return (
    <View
      style={[StyleSheet.absoluteFill, { overflow: 'hidden', pointerEvents: 'none' }]}
      pointerEvents="none"
    >
      <Svg style={StyleSheet.absoluteFill} width={W} height={H}>
        <Defs>
          {/* Pattern of 1px dots every 28px */}
        </Defs>
        {/* Render grid of circles manually for simplicity */}
        {Array.from({ length: Math.ceil(H / 28) }).map((_, row) =>
          Array.from({ length: Math.ceil(W / 28) }).map((__, col) => {
            // Fade based on distance from center-top
            const cx = col * 28, cy = row * 28;
            const dx = (cx - W / 2) / (W / 2);
            const dy = (cy - H * 0.3) / (H * 0.3);
            const dist = Math.sqrt(dx * dx + dy * dy);
            const opacity = Math.max(0, 0.06 * (1 - Math.min(dist / 1.2, 1)));
            if (opacity < 0.005) return null;
            return (
              <Circle
                key={`${row}-${col}`}
                cx={cx} cy={cy} r={1}
                fill={`rgba(255,255,255,${opacity.toFixed(3)})`}
              />
            );
          })
        )}
      </Svg>
    </View>
  );
}
```

```tsx
import React, { useEffect } from 'react';
import { View, StyleSheet } from 'react-native';
import Animated, { useSharedValue, useAnimatedStyle, withTiming, withSequence, runOnJS } from 'react-native-reanimated';
import { LinearGradient } from 'expo-linear-gradient';
import { AppLogo } from '../components/AppLogo';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { supabase } from '../lib/supabase';

export function SplashScreen({ navigation }: any) {
  const opacity = useSharedValue(0);
  const scale = useSharedValue(0.88);

  useEffect(() => {
    // Fade + scale in the logo
    opacity.value = withTiming(1, { duration: 600 });
    scale.value = withTiming(1, { duration: 600 });

    // After 1.5s, check session and route
    setTimeout(async () => {
      const onboardingShown = await AsyncStorage.getItem('onboarding_shown');
      const { data: { session } } = await supabase.auth.getSession();

      opacity.value = withTiming(0, { duration: 400 }, (done) => {
        if (done) {
          runOnJS(() => {
            if (session) {
              navigation.replace('Dashboard');
            } else if (onboardingShown) {
              navigation.replace('Auth');
            } else {
              navigation.replace('Onboarding');
            }
          })();
        }
      });
    }, 1500);
  }, []);

  const animStyle = useAnimatedStyle(() => ({
    opacity: opacity.value,
    transform: [{ scale: scale.value }],
  }));

  return (
    <LinearGradient
      colors={['#1A1040', '#0D0E1A', '#0B0C14']}
      locations={[0, 0.45, 1]}
      start={{ x: 0.5, y: 0 }}
      end={{ x: 0.5, y: 1 }}
      style={styles.screen}
    >
      <Animated.View style={[styles.center, animStyle]}>
        <AppLogo size={96} showHalo />
      </Animated.View>
    </LinearGradient>
  );
}

const styles = StyleSheet.create({
  screen: { flex: 1, alignItems: 'center', justifyContent: 'center' },
  center: { alignItems: 'center' },
});
```

## EXACT DESIGN RULES — MUST NOT DEVIATE

1. **Logo card background**: `LinearGradient colors={['#1C1F4A','#1A2E5A','#0F3B4A','#0D3040']}` — NOT flat teal, NOT blue-only
2. **Logo card border**: `borderColor: 'rgba(91,108,248,0.25)'` — subtle indigo, not white
3. **Logo card corner radius**: exactly `26px` (or `size * 0.27`)
4. **Screen background**: 3-layer system — deep space base + aurora glows + dot grid. Do NOT use a single flat dark color.
5. **No dark circle blob behind logo** — the logo floats on the layered background directly
6. **Font**: IBM Plex Arabic throughout. Do NOT fallback to System or SF Pro for Arabic text
7. **CTA button height**: `58px`, radius `18px`, gradient as specified
8. **Dot indicators**: inactive = 6×6px circle, active = 24×6px pill with accent glow shadow
9. **Active dot color**: `#5B6CF8` with `shadowColor: '#5B6CF8'` glow — not white, not grey
10. **Slide 3 button**: turns green `#10B981→#2DD4BF` gradient with green shadow — NOT still purple
11. **Particle system**: 8 particles, exact positions/colors/sizes listed above, float upward and fade
12. **Expanding rings**: positioned at screen bottom, `border: 1px solid rgba(91,108,248,0.08)`, 4s expand cycle staggered by 2s
13. **Animations**: enter animations use `cubic-bezier(0.16,1,0.3,1)` spring curve (in Reanimated: `Easing.out(Easing.back(1.7))`)
14. **Skip button**: top-left, pill-shaped, `rgba(255,255,255,0.05)` bg with border
15. **Language toggle**: top-right, same pill style, slightly brighter bg
